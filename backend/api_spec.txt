API should accept both application/json and application/x-www-form-urlencoded response and request content types.

/file/:data:

	GET
		Returns details of a file 

		responses:
			200:
			{
				size: int, // Size of the file in bytes
				name: string, // Name of the file
				download: string, // URL to download the file
			}

			405:
			{
				error: "Method Not Allowed"
			}

			404:
			{
				error: "File Not Found"
			}

			500:
			{
				error: "Internal Server Error"
			}

			403:
			{
				error: "Invalid Authorization Credentials"
			}

		parameters:
			{
				password: string // Not Required | Password needed to access a password protected file. Only required if the file is password protected.
			}

	POST
		Uploads a file

		responses:
			204:
			{
				error: "No Content"
			}

			413:
			{
				error: "Request Entity Too Large"
			}
		
		parameters:
		{
			file: 
		}
			- file:
				type: file
					description: File to upload

POST /:file
Saves the contents of the file
  needs:
		file: binary string
		expires: date, default: now + 3 days
		auto_delete: boolean, default: false
		max_downloads: integer, default: 0 (infinite)
		key?: string
	returns:
		admin_key: string
		download_url: string

GET /:file/info
Returns the statistics of the file
	returns:
		expires: date
		size: integer
		downloads: integer


DELETE /:file
  needs:
		admin_key: string
	returns:
		success: boolean
		message: string
